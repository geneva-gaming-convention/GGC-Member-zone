def is_proprietary
  id_to_check = 0
  if (params[:user_id])
    id_to_check = params[:user_id].to_i
  elsif (params[:id])
    id_to_check = params[:id].to_i
  elsif (current_logged_user)
    id_to_check = current_logged_user.id
  end
  if current_logged_user != nil
    if (current_logged_user.id == id_to_check)
      return true
    end
  end
  return false
end
